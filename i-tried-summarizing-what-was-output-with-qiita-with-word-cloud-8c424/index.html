About Word Cloud

WordCloud is to select words with a high frequency of occurrence from the text and to display them in a size according to the frequency of occurrence of the words. Since there is a library for Python, I will also refer to the code here. http://amueller.github.io/word_cloud/index.html

import urllib.request

url = "https://qiita.com/sksk_go"
res = urllib.request.urlopen(url)
soup = BeautifulSoup(res,'html.parser')
# Rewritten for tag acquisition
name = soup.find_all("a",class_="u-link-unstyled TagList__label")
ret = []
for t in name:
    ret.append(t.text)

Processing with Word Cloud

I will refer to this article. [Python] Visualizing the night of the Galactic Railway with WordCloud! 

import MeCab
from wordcloud import WordCloud

data = open("data.txt","rb").read()
text = data.decode('utf-8')

mecab = MeCab.Tagger("-ochasen")
node = mecab.parseToNode(text)

data_text = []

while node:
    word = node.surface
    hinnsi = node.feature.split(",")[0]
    if hinnsi in ["verb", "adverb", "adjective", "noun"]:
        data_text.append(word)
    else:
        print("|{0}| The part of speech of {1} is {1} so it is not added.".format(node.surface,node.feature.split(",")[0]))
        print("-"*35)
    node = node.next

text =''.join(data_text)
#Exclusion word
stop_words = [u'teru', u'teru', u'naru', u'teru', u'teru', u'teru', u'koto', u'kore', u'san', u' do it', \
             u'give', u'do', u'give', u'yes', u'make', u'do', u'think', \
             u'it', u'here', u'chan', u'kun', u'', u'te',u' to',u'',u' is',u''s',u 'Is', u'and', u'was', u'is', u'in', \
             u'not', u'also', u'na', u'i', u'or', u'because', u'like', u'']
wordcloud = WordCloud(font_path='/System/Library/Fonts/Hiragino Mincho ProN.ttc',width=480, height=300,background_color='white',stopwords=set(stop_words))
# Generate a word cloud from text.
wordcloud.generate(text)
# Save to file.
wordcloud.to_file('wordcloud.png')
