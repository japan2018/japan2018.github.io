"""

class OthelloCls:
    def __init__(self):
        self.ot_bit = list()
        #Initial setting = put four Othello pieces on the board
        for i in range(64):
            if i == 27 or i == 36:
                self.ot_bit.append('○')
            elif i == 28 or i == 35:
                self.ot_bit.append('●')
            else:
                self.ot_bit.append('・')
        self.ot_offdef ='●' # Which turn
        self.ot_search ='○' # The opposite hand
        # Use a dictionary to calculate adjacent frames in eight directions
        # Make a calculation table (direction is key, calculation value is defined by Tupple)
        self.ot_direction = dict() # Eight direction lookup table
        self.ot_direction = ('ul' :( -1, ,-1) ,# diagonally left up
                              'up' :( 0 ,-1) ,# right above
                              'ur' :( +1 ,-1) ,# diagonally above right
                              'lt': (-1 ,0) ,# left
                              'rt' :( +1 ,0) ,# right
                              'dl': (-1 ,+1) ,# diagonally below left
                              'dn' :( 0 ,+1) ,# right under
                              'dr' :( +1 ,+1 )} # diagonally below right
        # The end position when it is judged that it can be turned over
        self.ot_lastposX = 0 # end X to be turned over
        self.ot_lastposY = 0 # end Y that turns over

    # Input
    def ot_inputXY(self,otstr):
        while True:
            myXY = input(otstr)
            if myXY == "":
                continue
            # I added some check methods in the comment section! '20/5/30
            if myXY.isdigit() and (1 <= int(myXY) <= 8):
                return int(myXY)
            print("bubu~! Please input 1~8!")

    # It's your turn display
    def ot_yourturn(self):
        print("Hands of {}!".format(self.ot_offdef))

    # It's your turn
    def ot_changeturn(self):
        self.ot_offdef ='●' if self.ot_offdef =='○' else'○'
        self.ot_search ='●' if self.ot_offdef =='○' else'○'

    # Determine if it's over (change turn if you can't just put your hand in it)
    def ot_checkendofgame(self):
        #There is no place to put your own hand, and you can not put your opponent's hand either
        if 0> self.ot_canplacemypeace():
            self.ot_changeturn()
            if 0> self.ot_canplacemypeace():
                bc = myot.ot_bit.count('●')
                wc = myot.ot_bit.count('○')
                if (bc-wc)> 0:
                    bws = "● victory"
                elif (bc-wc) <0:
                    bws = "○ victory"
                else:
                    bws = "draw"
                print("{}. Thanks for your hard work!".format(bws))
                return 1
            else:
                print("There is no place to put {}. It will change to the opponent's turn!!".format(self.ot_search))
                return -1
        else:
            return 0

    # Find out if there is a place to put your hand
    def ot_canplacemypeace(self):
        for n in range(64):
            if self.ot_bit[n] !='・':
                continue
            for d in self.ot_direction:
                if 1 == self.ot_next_onepeace(int(n%8)+1,int(n/8)+1, d):
                    return 0
        # Come here if there is no place to put one
        return -1

    # Put your hand on the board of Othello
    def ot_place(self, ot_x, ot_y):
        ot_change = False
        for d in self.ot_direction:
            if 1 == self.ot_next_onepeace(ot_x,ot_y,d):
                self.ot_peace(ot_x,ot_y,self.ot_offdef)
                self.ot_changepeace(ot_x,ot_y,d)
                ot_change = True
        # If there is no valid direction
        if not ot_change:
            print("Could not place {}".format(self.ot_offdef))
            return -1
        # Display the screen and change hands to the opponent's turn
        self.ot_display()
        self.ot_changeturn()
        return 0

    #Process to put the next hand (even if it cannot be put)
    def ot_next_onepeace(self,ot_x,ot_y,ot_dir):
        #When the next hand in all directions from your position is the opposite hand and your own
        nx = self.ot_direction[ot_dir][0]
        ny = self.ot_direction[ot_dir][1]

        #Judgment that you can not put the next move in the first place
        if (nx <0 and ot_x == 1 )or (ny <0 and ot_y == 1 )or (nx == 1 and ot_x == 8 )or (ny == 1 and ot_y == 8 ):
            return -1

        # Get the next one (the left and the top are negative from your hand)nextpeace = self.ot_peace(ot_x+nx,ot_y+ny)

        # If the next hand is...
        if nextpeace =='・' or nextpeace == self.ot_offdef:
            return -1

        #Determine if there is a neighbor next door
        cx = ot_x+(nx*2)
        cy = ot_y+(ny*2)

        # It is judged that it can not be placed unless there is a neighbor next to it (if the direction is left or top, the left end and the top end are judged)
        if (nx <0 and cx == 0 )or (nx> 0 and cx == 9 )or (ny <0 and cy == 0 )or (ny> 0 and cy == 9 ):
            return -1

        # Look for it because you can take the next
        nextnextpeace = self.ot_peace(cx,cy)

        if nextnextpeace =='・':
            return -1 # I can't turn it over Notice

        if nextnextpeace == self.ot_offdef:
            # Record the end position that can be turned over
            self.ot_lastposX = cx
            self.ot_lastposY = cy
            return 1 # I can turn it over Notice

        #If your hand and your hand continue, call your function again (recursive)
        return self.ot_next_onepeace(ot_x+nx, ot_y+ny, ot_dir)

    # Hand over process
    def ot_changepeace(self,ot_x,ot_y,ot_dir):
        #When the next hand in all directions from your position is the opposite hand and your own
        nx = self.ot_direction[ot_dir][0]
        ny = self.ot_direction[ot_dir][1]
        # Rewrite the next one
        nextpeace = self.ot_peace(ot_x+nx,ot_y+ny,self.ot_offdef)
        # Next to next coordinate
        cx = ot_x+(nx*2)
        cy = ot_y+(ny*2)
        # End if the flipped position is the last coordinate
        if cx == self.ot_lastposX and cy == self.ot_lastposY:
            return
        # Call yourself again if you can still turn it over (recursive)
        return self.ot_changepeace(ot_x+nx, ot_y+ny, ot_dir)

    #Get the hand at the specified position (and rewrite it)
    def ot_peace(self,ot_x,ot_y,ot_chr=None):
        if ot_chr != None:
            self.ot_bit[(ot_y-1) * 8 + (ot_x-1)] = ot_chr
        return self.ot_bit[(ot_y-1)*8+(ot_x-1)]

    #Display the Othello board
    def ot_display(self):
        print("X ① ② ③ ④ ⑤ ⑥ ⑦ ⑧")
        for l in range(1,9):
            print("{}".format(l), end='')
            for c in range(1,9):
                print(" {}".format(self.ot_bit[(l-1)*8+(c-1)]), end='')
            print()
        print("○:{}●:{}".format(self.ot_bit.count('○'),self.ot_bit.count('●')))

if __name__ =='__main__':

    myot = OthelloCls()
    myot.ot_display()

    while True:
        #Determine if the game is over or where you can put your hand
        sts = myot.ot_checkendofgame()
        if sts == 1: # end the game
            break
        elif sts <0: # I can't put my hand
            continue

        # Show which turn
        myot.ot_yourturn()
        # Enter X and Y coordinates
        myx = myot.ot_inputXY("X = ")
        myy = myot.ot_inputXY("Y = ")
        # Put your hand at the specified coordinates
        myot.ot_place(myx,myy)

Learning Python today = Variables

In this chapter, I would like to learn about Python variables.

What is a variable? It’s not a “funny number”. Think of it as a container for the various values that your program generates. For example, when cooking, you cut the ingredients and put them in a bowl, which plays the same role as the variable. Those who cook do not add salt, sugar, and soy sauce to the cooking pot, but once in a measuring cup, half a teaspoon of salt, one teaspoon of sugar, three tablespoons of soy sauce, etc. You put them in and taste them, then put them in a pot. If you think of this measuring cup as a variable, there is no doubt.

Consider why a variable is needed. As you can see from the recipe, you can understand, but if you have a program that only makes curry for three people, you have decided what to put in, and you have decided the amount, too? But what if you wanted to make a curry-making program for as many people as possible? First of all, the number of people changes every time, so we have to put the specified number of people in the variable. Since the amount of ingredients such as potatoes also changes, this must be included in the variable. In reality, if a program that can make anything is better than a program that makes only curry, the number of types of ingredients, the number of seasonings to be added, and the quantity will change rapidly. Do you see that variables are indispensable for programs? ?

General variables and operations

As for general variables, of course the value of the variable can be rewritten.

If you say age = 12, you are assigning the value 12 to the variable named age. When this age 12 person gets one year old next year, the value in age can be increased by 1 by writing age += 1. This is called calculation. The following operators can be used to calculate. In addition, there is a priority in the calculation, and unless specified with (), the calculation is performed according to the order of occurrence and priority assigned to the operator.

Variable called array

It is sufficient to put only one value in one variable, but there are cases where you want to put a large number of values in one variable. In this case, use a variable called array. For example, if you want to have 7 days of the week in letters,

WeekDay = ["Monday" ,"Tuesday" ,"Wednesday" ,"Thursday" ,"Friday" ,"Saturday" ,"Sunday"]

When accessing the array value, specify WeekDay[0] and a numerical value called a subscript or index. Specify 0 for the index to access the very first value. It’s 0 on Monday, 1 on Tuesday.

Array has the concept of multidimensional array. In the previous example, it is a one-dimensional array containing seven character strings, but if you want to have calendar data, for example, you can create an array with 31 arrays and 12 arrays, so you can use Calendar[12][ You can also create a two-dimensional array called 31]. It is also handy to use a multidimensional array according to the program to be created, but it is incomprehensible even if it is too multidimensional, so it is good to use it with emphasis on being able to understand and writing clean code.
