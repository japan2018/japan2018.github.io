Morphological analysis

Refer to the following articles to prepare the necessary tools. Using mecab on Mac. -Qiita

Run the following code to split your review.

# coding: utf-8
import MeCab

mecab = MeCab.Tagger ('-d /usr/local/lib/mecab/dic/mecab-ipadic-neologd')

INPUT_FILE_PATH = "./scraping.csv"
OUTPUT_FILE_PATH = "./mecab.txt"

with open(INPUT_FILE_PATH) as f:
    text = f.read()

# mecab.parse('')# prevent string from being GC
node = mecab.parseToNode(text)
while node:
    #Get the word
    word = node.surface
    #Get part of speech
    pos = node.feature.split(",")[1]
    tmp_str ='{0} ,{1}\n'.format(word, pos)

    with open(OUTPUT_FILE_PATH, mode='a') as f:
        f.write(tmp_str)
    
    # Advance to the next word
    node = node.next

If you leave it as it is, only words that have no meaning, such as particles, will be extracted, so extract the part of speech that is likely to have meaning as a word.

$ grep -e "noun" -e "verb" -e "adjective" mecab.txt | grep -v "pronoun" | cut -d',' -f 1 >mecab_edited.txt

from wordcloud import WordCloud

FONT_PATH = "/System/Library/Fonts/Hiragino Kaku Gothic W9.ttc"

INPUT_FILE_PATH = "./mecab_edited.txt"
OUTPUT_FILE_PATH = "./wordcloud.png"

with open(INPUT_FILE_PATH) as f:
    text = f.read()

stop_words = ["From output image", "I want to remove", "Words", "Set"]

wordcloud = WordCloud(background_color="white",
    font_path=FONT_PATH,
    width=800,height=600,
    stopwords=set(stop_words)).generate(text)

wordcloud.to_file(OUTPUT_FILE_PATH)
