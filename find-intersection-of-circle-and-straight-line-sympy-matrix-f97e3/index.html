# [Example] Find the intersection of the circle with center (3, 2) radius 5 and the line 3x+2y-16=0
from sympy import *
var('x0 y0 r x1 y1 x2 y2 c s tx ty d sx sy')
x0=3
y0=2
r=5
y1=0
v=solve([3*x1+2*y1-16])
x1=v[x1]
x2=0
v=solve([3*x2+2*y2-16])
y2=v[y2]
d=sqrt((x1-x2)**2+(y1-y2)**2)
v=solve((c*0-s*0+tx-x1,s*0+c*0+ty-y1,c*ds*0+tx-x2,s*d+c*0+ty-y2 ],[c,s,tx,ty])
A=Matrix((
        [v[c],-v[s],v[tx]],
        [v[s], v[c],v[ty]],
        [0, 0, 1]
])
B=Matrix((
       [x0],
       [y0],
       [1]
])
AinvB=A.inv()*B
sx=AinvB[0].subs([(c**2 + s**2,1)])
sy=AinvB[1].subs([(c**2 + s**2,1)])
It's a sequel.
B=Matrix((
         [sx+sqrt(r**2-sy**2)],
         [0],
         [1]
])
AB=A*B
print(float(AB[0]),float(AB[1]))
B=Matrix((
         [sx-sqrt(r**2-sy**2)],
         [0],
         [1]
])
AB=A*B
print(float(AB[0]),float(AB[1]))
