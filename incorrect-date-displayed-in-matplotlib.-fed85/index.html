#3 Code to display date correctly on the horizontal axis

It is necessary to change the date line of the data frame from datetime to Matplotlib date.

Add dat[‘DATE’] = mdates.date2num(dat[‘DATE’]).

import datetime
import matplotlib.pyplot as plt
import matplotlib.dates as mdates
import seaborn as sns
import pandas as pd

#Define the Data Set. (Date is described as datetime.datetime. Notated as date type)
dat = [
    [datetime.datetime(2020,1,1),4,10],
    [datetime.datetime(2020,1,2),7,7],
    [datetime.datetime(2020,1,3),10,4],
    [datetime.datetime(2020,1,4),13,2],
    [datetime.datetime(2020,1,5),17,1],
    [datetime.datetime(2020,1,6),12,4],
    [datetime.datetime(2020,1,7),9,3],
    [datetime.datetime(2020,1,8),7,8],
    [datetime.datetime(2020,1,9),5,9],
    [datetime.datetime(2020,1,10),3,12],
     
]

dat=pd.DataFrame(dat,columns=["DATE","Y","Z"])
dat['DATE'] = mdates.date2num(dat['DATE']) # add here Convert datetime objects to Matplotlib dates.
dat.set_index("DATE",inplace=True) #Set the date displayed on the horizontal axis to the DataFrame index.
print(dat)


fig = sns.mpl.pyplot.figure() # Create an object that represents the graph.
ax=dat.plot(marker="o",figsize=(15, 5))
ax.legend() # render the legend

# Graph format setting (Set the date display method on the horizontal axis.)
days = mdates.DayLocator(bymonthday=None, interval=2, tz=None) # Horizontal axis: Display "every day". (Duplicate date is displayed without this line)
daysFmt = mdates.DateFormatter('%Y-%m-%d') # Horizontal axis: Format Y-M-D.
ax.xaxis.set_major_locator(days) # Display the date on the horizontal axis.
ax.xaxis.set_major_formatter(daysFmt) # Display the date on the horizontal axis.
fig.autofmt_xdate() # Make the horizontal axis date slanted so that you can see it easily.

# Name the graph
ax.set_xlabel('Date') # set X axis title
ax.set_ylabel('Y') # set Y axis title
plt.title(r"TEST",fontname="MS Gothic") # Set the graph title. When specifying Japanese, font name must be specified.
# Set graph size
fig.set_figheight(10)
fig.set_figwidth(20)
#Set the display range of the horizontal axis
ax.set_xlim(datetime.datetime(2020,1,1), datetime.datetime(2020,1,12))
