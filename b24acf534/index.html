Any function notes

In the following example, a process with high calculation cost (heavy_task) is unnecessarily executed 3 times.

# Input
def heavy_task(ret):
    print(ret)
    return ret


if any([heavy_task(True), heavy_task(False), heavy_task(False)]):
    print('Complete')

# Output
True
False
False
Complete

In this case, it is more efficient to use if.

# Input
def heavy_task(ret):
    print(ret)
    return ret


if heavy_task(True) or heavy_task(False) or heavy_task(False):
    print('Complete')

# Output
True
Complete

I find defaultdict especially useful when creating some sort of aggregation. The process is simple because you do not have to confirm the existence of each key.

# Input
import collections

products = ['Notes','Pencils','Kesigomu']

sales = [
    ['Notes', 1],
    ['Kishigomu', 2],
    ['Kishigomu', 1],
]

aggregates = collections.defaultdict(int)

for product, cnt in sales:
    aggregates[product] += cnt

print('--- Today's sales record ---')

for product in products:
    print('%s: %d'% (product, aggregates[product]))

# Output
- -- Today's sales record ---
Note: 1 piece
Pencil: 0
Kashigomu: 3

You can also implement a multi-associative array simply by using the property that any function (callable) can be passed.

# Input

nested = defaultdict(lambda: defaultdict(int))

nested['a']['a'] += 1
nested['a']['a'] += 1
nested['a']['b'] += 1
nested['b']['c'] += 1

print(json.dumps(nested))

{"a": {"a": 2, "b": 1}, "b": {"c": 1}}

tuple

A tuple is similar to an array, but it cannot be reassigned because it is immutable.

Since it is immutable, it can be used as an associative array key.

yamada = ('Yamada','Taro') # This is a tuple
tanaka = ('Tanaka','Jiro') # This is also a tuple

print(yamada[0]) # -> Yamada

yamada[0] ='Tanaka' # TypeError:'tuple' object does not support item assignment

ages = {
    yamada: 20,
    tanaka: 18,
}

print(ages[tanaka]) # -> 18

By using this characteristic, grouping of similar data can be performed efficiently.

# Input
from collections import defaultdict

data = [
    {'circle':'baseball','name':'yamada','age': 10},
    {'circle':'baseball','name':'sato','age': 10},
    {'circle':'baseball','name':'suzuki','age': 11},
    {'circle':'tennis','name':'tanaka','age': 10},
]

per_circle_age = defaultdict(list)

for v in data:
    k = (v['circle'], v['age']) # generate a tuple named (circle name, age)
    per_circle_age[k].append(v['name']) # aggregate using tuple as key

for (circle, age), members in per_circle_age.items():
    print(f'{circle} {age} year old member: {members}')

# Output
"A 10 year old member of baseball: ['yamada','sato']"
"11-year-old member of baseball: ['suzuki']"
"10-year-old member of tennis: ['tanaka']"

dataclasses package

This function is available from Python 3.7.

It’s convenient because you don’t have to write assignment processing like self.name = name in the constructor.

import dataclasses


# frozen=True treats it as an immutable object
@dataclasses.dataclass(frozen=True)
class User:
    last_name: str
    first_name: str
        
    def full_name(self):
        return f'{self.last_name} {self.first_name}'
        
        
yamada = User(last_name='Yamada', first_name='Taro')
tanaka = User(last_name='Tanaka', first_name='Jiro')

yamada.full_name() # -> Yamada Taro

# Easy to convert to associative array
dataclasses.asdict(yamada) # -> {'last_name':'Yamada','first_name':'Taro'}

#Comparable
yamada2 = User(last_name='Yamada', first_name='Taro')
yamada == yamada2 # -> True
yamada == tanaka # -> False
yamada in [yamada2] # -> True

If "#frozen=True" is set, the value cannot be reassigned.
yamada.last_name ='Sato' # -> FrozenInstanceError: cannot assign to field'last_name'

#immutable so it can be used as an associative array key
d = {yamada:'foo', tanaka:'bar'}

#Set operation (Set type) is also possible
{yamada, tanaka} & {yamada2} # -> {User(last_name='Yamada', first_name='Taro')}
