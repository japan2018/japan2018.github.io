#Introduction

#Content Contents of Chapter 3 of “Introduction to Control Engineering with Python”

common

Import the following libraries

import sympy import control

Transfer function

Code below

# chapter 3-2 Transfer function model
Np = [0,1]
Dp = [1,2,3]
P = control.tf(Np, Dp)
print(P)

output
      1
- ------------
s^2 + 2 s + 3

State space

Code below

#3-3 State space model
A = '1 1 2; 2 1 1; 3 4 5'
B = '2; 0; 1'
C = '1 1 0'
D = '0'
P = control.ss(A,B,C,D)
print(P)
sysA, sysB, sysC, sysD = control.ssdata(P)
print(sysA)

output
