You can see that the derived class of #GeneratedClass is MyClass.
class MyClass(GeneratedClass):
#First, __init__(self) is called when the behavior is loaded.
    def __init__(self):
        You can see that you are overriding __init__(self) of #GeneratedClass.
        GeneratedClass.__init__(self)
        #Let's put here what is called when the behavior is loaded.

    def onLoad(self):
        #put initialization code here
        #Let's put the initialization process here.
        Called after #behavior has been loaded and all init has been called.
        pass

    def onUnload(self):
        #put clean-up code here
        #Called when there is an input in onInput_onStop when the application ends.
        pass

    def onInput_onStart(self):
        #self.onStopped() #activate the output of the box
        It is called when there is input in #onStart.
        Please note that if you do not write #self.onStopped(), output will not be made.
        pass

    def onInput_onStop(self):
    It is called when there is input to #onStop.
        self.onUnload() #it is recommended to reuse the clean-up as the box is stopped
        self.onStopped() #activate the output of the box
