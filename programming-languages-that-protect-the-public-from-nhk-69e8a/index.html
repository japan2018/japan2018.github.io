#Combinator logic The grammar that protects the people from NHK can unambiguously describe a binary tree that has two types of leaves, NHK and people.

As a programming language with such a simple syntax, I focused on Lazy K.

Lazy K combinator logic based programming language, you can write a program with only three combinators S K I. Of these, I is known to be extensibly equal to S K K, so the only effective combinators are S and K. 1

Therefore, a Turing-complete language can be created by simply replacing the following 2.

Grammar to protect people from NHK Combinator
NHK S
People K

#!/usr/bin/env python3
import sys
class Combinator:
    '''X-Y is synonym of Apply(X,Y)'''
    def __sub__(self, arg):
        return Apply(self, arg)

    def evaluate(self):
        return self

    def get_left(self, xs):
        xs.append(self)

class _SIN(Combinator):
    '''Do not call _SIN() .Use SIN instead'''
    sinlist = None
    def evaluate(self):
        if _SIN.sinlist is None:
            _SIN.sinlist = str_to_sk(sys.stdin.read())
        return _SIN.sinlist

class _S(Combinator):
    '''Do not call _S() .Use S instead'''

class _K(Combinator):
    '''Do not call _K() .Use K instead'''

class Foreign(Combinator):
    '''Non-combinator class.((Foreign f)-(Foreign x)).evaluate() == f(x)'''
    def __init__(self, val):
        self.val = val

class Apply(Combinator):
    '''Application of combinators. Apply(X,Y) represents (X Y)'''
    def __init__(self, fun, arg):
        self.fun = fun
        self.arg = arg

    def get_left(self, xs):
        xs.append(self.arg)
        self.fun.get_left(xs)

    def evaluate(self):
        left_evaled = self.fun.evaluate()-self.arg
        left = []
        left_evaled.get_left(left)
        n = len(left)
        if(n == 4 and left[3] == S):
            return (left[2]-left[0]-(left[1]-left[0])).evaluate()
        if(n == 3 and left[2] == K):
            return (left[1]).evaluate()
        if(n == 2 and isinstance(left[1], Foreign)):
            return Foreign(left[1].val(left[0].evaluate().val))
        return left_evaled

def str_to_sk(string):
    ret = cons-int_to_church(256)-false
    for c in reversed(string):
        ret = cons-int_to_church(ord(c))-ret
    return ret

def church_to_int(c):
    return (c-(Foreign(lambda n: n+1))-(Foreign(0))).evaluate().val

def int_to_church(n):
    if(n == 0):
        return zero
    if(n == 1):
        return one
    if(n == 2):
        return two
    else:
        ret = one
        for c in bin(n)[3:]:
            ret = mul-two-ret
            if c == '1':
                ret = succ-ret
        return ret

def get_car(c):
    return (c-true).evaluate()

def get_cdr(c):
    return (c-false).evaluate()

def parseNHK(code):
    len_NHK = len('NHK')
    len_people = len ('people')
    len_from = len('from')
    len_save = len('Protecting party')

    stk = []
    code = code.replace('\n', ``)

    while code:
        if code.startswith('NHK'):
            stk.append(S)
            code = code[len_NHK:]
        elif code.startswith('national'):
            stk.append(K)
            code = code[len_people:]
        elif code.startswith('to'):
            code = code[len_from:]
        elif code.startswith('Protecting Party'):
            try:
                arg = stk.pop()
                fun = stk.pop()
            except IndexError as e:
               raise Exception('parseNHK failed. Too much "The party that protects"")
            stk.append(fun-arg)
            code = code[len_save:]
        elif code.startswith('') or code.startswith('\t') or code.startswith('''):
            code = code[1:]
        else:
            raise Exception('parseNHK failed at'+ code[:10])
    if len(stk) == 1:
        return stk.pop()
    else:
        raise Exception('parseNHK failed. Too few "to protect the party"')
