from random import choices #choice → choices


play = input(Type'play and you can draw your omikuji:')

while True:
    if play =='play':
        break

    print('Please enter again.')
    play = input(Type'play and you can draw your omikuji:')

while True:
    KUJI = ['Daikichi','Nakichi','Okichi','Sueyoshi','Kyou','Kyokichi']
    print(choices(KUJI, weights=[1, 5, 10, 10, 5, 1])[0]) # Changed here.
    continue_ = input('Do you want to try again? [y/n]:')
    while True:
        if continue_ =='y' or continue_ =='n': # The code here and there is also arranged.
            break
        else:
            continue_ = input('Please enter y or n:')

    if continue_ =='n':
        break
