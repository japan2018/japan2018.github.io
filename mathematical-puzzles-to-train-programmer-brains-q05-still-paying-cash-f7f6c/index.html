Mathematical puzzles to train programmer brains Q05 Still paying cash?

Apr 22, 2020 Python numpy itertools combination 

# Problem overview

Exchange 1000 yen into a combination of 500 yen coin, 100 yen coin, 50 yen coin and 10 yen coin. However, the total number of coins is 15 or less.

#Code Solve without thinking

#Q05 Still cash payment?

combinations = [] # list of combinations
for c500 in range(3):
    for c100 in range(11):
        for c50 in range(16):
            for c10 in range(16):
                if c500 + c100 + c50 + c10 <= 15:
                    if c500 * 500 + c100 * 100 + c50 * 50 + c10 *10 == 1000:
                        combinations.append((c500, c100, c50, c10))
                
print(combinations)
print(len(combinations))

#Code part 2 Write with as few for statements as possible

import itertools
import numpy as np

combinations = []
for c in list(itertools.product(list(range(16)), repeat=4)):
    m = np.array(list(c)) * np.array([500, 100, 50, 10])
    if np.sum(c) <= 15 and np.sum(m) == 1000:
        combinations.append(c)
        
print(combinations)
print(len(combinations))

##itertools.product Find the Cartesian product of the arrays. You can use the repeat option to find the Cartesian product with yourself. Returns an itertools object but can be cast to a list. Example of use

import itertools

x = ['a','b']
y = ['c','d']
z = list(range(3)) #[0, 1, 2]

print(list(itertools.product(x, y)))
#[('a','c'), ('a','d'), ('b','c'), ('b','d')]

print(list(itertools.product(x, repeat=2)))
#[('a','a'), ('a','b'), ('b','a'), ('b','b')]

print(list(itertools.product(z, repeat=2)))
#[(0, 0), (0, 1), (0, 2), (1, 0), (1, 1), (1, 2), (2, 0), (2, 1), (twenty two)]

##numpy list arithmetic In a regular list, the + sign represents a concatenation of the lists. Any other operation symbol is an error. numpy returns a list of the results of operations on the elements at the same position when the lists are connected by operation symbols.

import numpy as np

x = [1, 3, 5]
y = [2, 4, 6]

print(x + y) #[1, 3, 5, 2, 4, 6]
print(x-y) #error
print(x * y) #error
print(x / y) #error

print(np.array(x) + np.array(y)) # [3 7 11]
print(np.array(x)-np.array(y)) # [-1 -1 -1]
print(np.array(x) * np.array(y)) # [2 12 30]
print(np.array(x) / np.array(y)) # [0.5 0.75 0.83333333]
