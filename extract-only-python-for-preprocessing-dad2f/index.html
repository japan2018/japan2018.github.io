#Function to convert month numbers to seasons
def to_season(month_num):
  season ='winter'
  if 3 <= month_num <= 5:
    season ='spring'
  elif 6 <= month_num <= 8:
    season ='summer'
  elif 9 <= month_num <= 11:
    season ='autumn'
  
  return season

#Convert to season
reserve_tb['reserve_season'] = pd.Categorical(
  reserve_tb['reserve_datetime'].dt.month.apply(to_season),
  categories=['spring','summer','autumn','winter']
)
