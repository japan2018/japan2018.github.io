## 09. Typoglycemia

Create a program that leaves the leading and trailing characters of each word in a word string separated by spaces and rearranges the order of the other characters randomly. However, words with a length of 4 or less are not sorted. Give an appropriate English sentence (for example, “I couldn’t believe that I could actually understand what I was reading :the phenomenal power of the human mind .") and confirm the execution result.

###Go

package main

import (
"fmt"
"math/rand"
"strings"
"time"
)

/*
 * Randomly fetch one element from the array and return the value, and delete the array
 */
func choice(m *[]string) string {
// Generate a random number within the number of values
i := rand.Intn(len(*m))

// Get the word and delete it from the map
str := (*m)[i]

// This is convenient (*reference)
*m = append((*m)[:i],(*m)[i+1:]...)

return str
}


func main() {
body := "I couldn't believe that I could actually understand what I was reading :the phenomenal power of the human mind ."
var randWords []string
fixWords := map[int]string{} // I want to leave a fixed position, so use the position as a subscript of Map

// random number initialization
rand.Seed(time.Now().UnixNano())

// Divide into words and get the number of characters
words := strings.Split(body, "")
length := len(words)

// Classify fixed position and variable position words
for i, word := range words {
// fixed position for beginning and end and words less than 4 characters
if i == 0 || i == (length-1) || len(word) <= 4 {
fixWords[i] = word
} else {
randWords = append(randWords,word)
}
}

// Set word to fixed position or variable (random number) position while looping the number of words
var result [] string
for i:=0;i<length;i++ {
if word, ok := fixWords[i]; ok {
result = append(result, word)
} else {
result = append(result, choice(&randWords))
}
}

// result display
fmt.Println(strings.Join(result, ""))
}

# -*- coding: utf-8 -*-
import random

src = "I couldn't believe that I could actually understand what I was reading :the phenomenal power of the human mind ."
fixWords = {}
randWords = []

# Divide into words and classify fixed position and variable position words
words = src.split(" ")
for i,word in enumerate(words):
    # Beginning and ending and words with less than 4 characters are fixed positions
    if i == 0 or i == (len(words)-1) or len(word) <= 4:
        fixWords[i] = word;
    else:
        randWords.append(word)

# Shuffle inside array
random.shuffle(randWords)

# Set a word to a fixed position or variable (random number) position while looping the number of words
result = []
for i in range(len(words)):
    if i in fixWords:
        result.append(fixWords[i])
    else:
        result.append(randWords.pop(0))

# Combine results with blanks
print''.join(result)

function choice() {
    // Generate a random number within the number of values and get the corresponding word
    var i = Math.floor(Math.random() * randWords.length);
    var s = randWords[i]

    // Get the corresponding word, delete it from the map and rearrange the map
    delete randWords[i];
    randWords = randWords.filter(function (el) {return el; }); // The map doesn't shrink even if you delete it

    return s;
}
